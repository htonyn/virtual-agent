CREATE TABLE IF NOT EXISTS customers(
    customerid int(6) AUTO_INCREMENT,
    firstname varchar(255),
    lastname varchar(255),
    address varchar(255),
    PRIMARY KEY (customerid)
);

CREATE TABLE IF NOT EXISTS orders(
    orderid int(12) AUTO_INCREMENT,
    customerid int(6),
    date_purchased DATE,
    PRIMARY KEY (orderid)
);

CREATE TABLE IF NOT EXISTS order_items(
    orderid int(12),
    productid int(12),
    produce_quantity int(2),
    product_price int,
    PRIMARY KEY (orderid)
);

CREATE TABLE IF NOT EXISTS users(
    userid int(6) AUTO_INCREMENT,
    login varchar(255),
    password varchar(255),
    customerid int(6),
    admin_flag TINYINT(1),
    PRIMARY KEY (userid)
);

CREATE TABLE IF NOT EXISTS shoppingcart(
    cartid int(6),
    customerid int(6),
    date_initialized DATE,
    PRIMARY KEY (cartid)
);

CREATE TABLE IF NOT EXISTS shoppingcart_items(
    cartid int(6),
    productid int(12),
    product_quantity int(2),
    product_price int
);
